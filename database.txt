-- ============================================================
-- 1. ACCOUNTS SCHEMA — Identity & Security
-- ============================================================

CREATE TABLE IF NOT EXISTS accounts.users (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    email           VARCHAR UNIQUE NOT NULL,
    password_hash   TEXT,
    full_name       VARCHAR,
    google_auth_id  VARCHAR UNIQUE,
    role            VARCHAR NOT NULL DEFAULT 'general'
                        CHECK (role IN ('rehab', 'general', 'fitness')),
    is_active       BOOLEAN NOT NULL DEFAULT TRUE,
    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS accounts.admins (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    username        VARCHAR NOT NULL,
    email           VARCHAR UNIQUE NOT NULL,
    password_hash   TEXT NOT NULL,
    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS accounts.password_resets (
    id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id     UUID NOT NULL REFERENCES accounts.users(id) ON DELETE CASCADE,
    token       VARCHAR NOT NULL,
    expires_at  TIMESTAMPTZ NOT NULL
);

-- Indexes
CREATE INDEX IF NOT EXISTS idx_users_email          ON accounts.users(email);
CREATE INDEX IF NOT EXISTS idx_users_google_id      ON accounts.users(google_auth_id);
CREATE INDEX IF NOT EXISTS idx_pwd_reset_user       ON accounts.password_resets(user_id);
CREATE INDEX IF NOT EXISTS idx_pwd_reset_token      ON accounts.password_resets(token);


-- ============================================================
-- 2. PROFILE SCHEMA — Personalized User Data
-- ============================================================

CREATE TABLE IF NOT EXISTS profile.health_profiles (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id             UUID UNIQUE NOT NULL REFERENCES accounts.users(id) ON DELETE CASCADE,
    age                 INT CHECK (age > 0 AND age < 130),
    weight              DECIMAL(5,2),   -- kg
    height              DECIMAL(5,2),   -- cm
    bmi                 DECIMAL(4,2),
    primary_goal        VARCHAR,        -- e.g. 'weight_loss', 'muscle_gain', 'rehab'
    injury_details      TEXT,
    recovery_stage      VARCHAR,        -- e.g. 'acute', 'subacute', 'chronic'
    medical_diet_notes  TEXT
);

CREATE TABLE IF NOT EXISTS profile.user_chronic_conditions (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    profile_id      UUID NOT NULL REFERENCES profile.health_profiles(id) ON DELETE CASCADE,
    condition_name  VARCHAR NOT NULL
);

-- Indexes
CREATE INDEX IF NOT EXISTS idx_health_profiles_user       ON profile.health_profiles(user_id);
CREATE INDEX IF NOT EXISTS idx_chronic_conditions_profile ON profile.user_chronic_conditions(profile_id);


-- ============================================================
-- 3. LIBRARY SCHEMA — Master Content
-- ============================================================

CREATE TABLE IF NOT EXISTS library.exercises (
    id                         UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    source_id                  VARCHAR UNIQUE,  -- original ID from JSON (e.g. "4F5866F8")
    title                      VARCHAR NOT NULL,
    exercise_type              VARCHAR,         -- e.g. 'weight_reps', 'reps_only', 'duration'
    muscle_group               VARCHAR,         -- primary muscle group
    equipment_category         VARCHAR,         -- e.g. 'barbell', 'dumbbell', 'machine', 'none'
    url                        TEXT,            -- video/resource URL
    media_type                 VARCHAR,         -- e.g. 'video', 'gif', 'image'
    thumbnail_url              TEXT,
    instructions               TEXT,
    manual_tag                 VARCHAR,         -- search/filter helper tag from source data
    priority                   INT DEFAULT 0,   -- display ordering
    is_custom                  BOOLEAN NOT NULL DEFAULT FALSE,
    is_archived                BOOLEAN NOT NULL DEFAULT FALSE,
    hundred_percent_bodyweight BOOLEAN NOT NULL DEFAULT FALSE
);

CREATE TABLE IF NOT EXISTS library.exercise_secondary_muscles (
    id           UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    exercise_id  UUID NOT NULL REFERENCES library.exercises(id) ON DELETE CASCADE,
    muscle_name  VARCHAR NOT NULL
);

CREATE TABLE IF NOT EXISTS library.meals (
    id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    title         VARCHAR NOT NULL,
    url           TEXT,
    image_url     TEXT,
    servings      INT CHECK (servings > 0),
    prep_time     VARCHAR(50),
    time_to_make  VARCHAR(50),
    instructions  JSONB,               -- array of step strings
    guide_info    TEXT,
    created_at    TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS library.meal_nutrition (
    meal_id         UUID PRIMARY KEY REFERENCES library.meals(id) ON DELETE CASCADE,
    calories_cal    INT,
    kilojoules_kj   INT,
    protein_g       DECIMAL(6,2),
    total_fat_g     DECIMAL(6,2),
    carbohydrates_g DECIMAL(6,2),
    sugar_g         DECIMAL(6,2)
);

CREATE TABLE IF NOT EXISTS library.meal_tags (
    id        UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    meal_id   UUID NOT NULL REFERENCES library.meals(id) ON DELETE CASCADE,
    tag_name  VARCHAR(100) NOT NULL
);

CREATE TABLE IF NOT EXISTS library.ingredients (
    id           UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    meal_id      UUID NOT NULL REFERENCES library.meals(id) ON DELETE CASCADE,
    description  TEXT NOT NULL
);

-- Indexes
CREATE INDEX IF NOT EXISTS idx_exercises_type     ON library.exercises(exercise_type);
CREATE INDEX IF NOT EXISTS idx_exercises_muscle   ON library.exercises(muscle_group);
CREATE INDEX IF NOT EXISTS idx_exercises_archived ON library.exercises(is_archived);
CREATE INDEX IF NOT EXISTS idx_sec_muscles_ex     ON library.exercise_secondary_muscles(exercise_id);
CREATE INDEX IF NOT EXISTS idx_meals_title        ON library.meals(title);
CREATE INDEX IF NOT EXISTS idx_meal_tags_meal     ON library.meal_tags(meal_id);
CREATE INDEX IF NOT EXISTS idx_ingredients_meal   ON library.ingredients(meal_id);

-- 4. PLANS SCHEMA — Structured Programs
-- ============================================================

CREATE TABLE IF NOT EXISTS plans.workout_plans (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    title               VARCHAR NOT NULL,
    difficulty_level    VARCHAR CHECK (difficulty_level IN ('beginner', 'intermediate', 'advanced')),
    created_by_admin_id UUID REFERENCES accounts.admins(id) ON DELETE SET NULL,
    created_by_user_id  UUID REFERENCES accounts.users(id) ON DELETE SET NULL,
    is_template         BOOLEAN NOT NULL DEFAULT FALSE,
    CHECK (
        (created_by_admin_id IS NOT NULL AND created_by_user_id IS NULL) OR
        (created_by_admin_id IS NULL     AND created_by_user_id IS NOT NULL)
    )
);

CREATE TABLE IF NOT EXISTS plans.workout_plan_items (
    id                UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    workout_plan_id   UUID NOT NULL REFERENCES plans.workout_plans(id) ON DELETE CASCADE,
    exercise_id       UUID NOT NULL REFERENCES library.exercises(id) ON DELETE RESTRICT,
    sets              INT CHECK (sets > 0),
    reps              INT CHECK (reps > 0),
    rest_time_seconds INT CHECK (rest_time_seconds >= 0)
);

CREATE TABLE IF NOT EXISTS plans.meal_plans (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    title               VARCHAR NOT NULL,
    goal_type           VARCHAR,        -- e.g. 'weight_loss', 'muscle_gain', 'maintenance'
    created_by_admin_id UUID REFERENCES accounts.admins(id) ON DELETE SET NULL,
    created_by_user_id  UUID REFERENCES accounts.users(id) ON DELETE SET NULL,
    is_template         BOOLEAN NOT NULL DEFAULT FALSE,
    CHECK (
        (created_by_admin_id IS NOT NULL AND created_by_user_id IS NULL) OR
        (created_by_admin_id IS NULL     AND created_by_user_id IS NOT NULL)
    )
);

CREATE TABLE IF NOT EXISTS plans.meal_plan_items (
    id           UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    meal_plan_id UUID NOT NULL REFERENCES plans.meal_plans(id) ON DELETE CASCADE,
    meal_id      UUID NOT NULL REFERENCES library.meals(id) ON DELETE RESTRICT,
    day_number   INT NOT NULL CHECK (day_number >= 1),
    meal_type    VARCHAR NOT NULL CHECK (meal_type IN ('breakfast', 'lunch', 'dinner', 'snack'))
);

CREATE TABLE IF NOT EXISTS plans.user_plan_enrollments (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id         UUID NOT NULL REFERENCES accounts.users(id) ON DELETE CASCADE,
    workout_plan_id UUID REFERENCES plans.workout_plans(id) ON DELETE SET NULL,
    meal_plan_id    UUID REFERENCES plans.meal_plans(id) ON DELETE SET NULL,
    status          VARCHAR NOT NULL DEFAULT 'active'
                        CHECK (status IN ('active', 'paused', 'completed', 'dropped')),
    enrolled_at     TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Indexes
CREATE INDEX IF NOT EXISTS idx_workout_plans_admin ON plans.workout_plans(created_by_admin_id);
CREATE INDEX IF NOT EXISTS idx_workout_plans_user  ON plans.workout_plans(created_by_user_id);
CREATE INDEX IF NOT EXISTS idx_workout_items_plan  ON plans.workout_plan_items(workout_plan_id);
CREATE INDEX IF NOT EXISTS idx_workout_items_ex    ON plans.workout_plan_items(exercise_id);
CREATE INDEX IF NOT EXISTS idx_meal_plans_admin    ON plans.meal_plans(created_by_admin_id);
CREATE INDEX IF NOT EXISTS idx_meal_plans_user     ON plans.meal_plans(created_by_user_id);
CREATE INDEX IF NOT EXISTS idx_meal_items_plan     ON plans.meal_plan_items(meal_plan_id);
CREATE INDEX IF NOT EXISTS idx_enrollments_user    ON plans.user_plan_enrollments(user_id);
CREATE INDEX IF NOT EXISTS idx_enrollments_status  ON plans.user_plan_enrollments(status);

-- 5. TRACKER SCHEMA — Live Activity
-- ============================================================

CREATE TABLE IF NOT EXISTS tracker.workout_sessions (
    id               UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id          UUID NOT NULL REFERENCES accounts.users(id) ON DELETE CASCADE,
    workout_plan_id  UUID REFERENCES plans.workout_plans(id) ON DELETE SET NULL,
    started_at       TIMESTAMPTZ,
    completed_at     TIMESTAMPTZ,
    status           VARCHAR NOT NULL DEFAULT 'in_progress'
                         CHECK (status IN ('in_progress', 'completed', 'abandoned')),
    CHECK (completed_at IS NULL OR completed_at >= started_at)
);

CREATE TABLE IF NOT EXISTS tracker.workout_session_items (
    id             UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    session_id     UUID NOT NULL REFERENCES tracker.workout_sessions(id) ON DELETE CASCADE,
    exercise_id    UUID NOT NULL REFERENCES library.exercises(id) ON DELETE RESTRICT,
    set_number     INT NOT NULL CHECK (set_number > 0),
    reps_completed INT CHECK (reps_completed >= 0),
    weight_used    DECIMAL(6,2) CHECK (weight_used >= 0),  -- kg
    is_completed   BOOLEAN NOT NULL DEFAULT FALSE
);

CREATE TABLE IF NOT EXISTS tracker.user_workout_schedule (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id         UUID NOT NULL REFERENCES accounts.users(id) ON DELETE CASCADE,
    workout_plan_id UUID REFERENCES plans.workout_plans(id) ON DELETE SET NULL,
    scheduled_date  DATE NOT NULL,
    is_completed    BOOLEAN NOT NULL DEFAULT FALSE
);

CREATE TABLE IF NOT EXISTS tracker.user_meal_schedule (
    id             UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id        UUID NOT NULL REFERENCES accounts.users(id) ON DELETE CASCADE,
    meal_id        UUID NOT NULL REFERENCES library.meals(id) ON DELETE RESTRICT,
    scheduled_date DATE NOT NULL,
    meal_type      VARCHAR NOT NULL CHECK (meal_type IN ('breakfast', 'lunch', 'dinner', 'snack')),
    is_eaten       BOOLEAN NOT NULL DEFAULT FALSE
);

CREATE TABLE IF NOT EXISTS tracker.daily_logs (
    id                 UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id            UUID NOT NULL REFERENCES accounts.users(id) ON DELETE CASCADE,
    date               DATE NOT NULL,
    calories_consumed  INT CHECK (calories_consumed >= 0),
    workouts_completed INT CHECK (workouts_completed >= 0),
    recovery_notes     TEXT,
    UNIQUE (user_id, date)  -- one log per user per day
);

-- Indexes
CREATE INDEX IF NOT EXISTS idx_sessions_user         ON tracker.workout_sessions(user_id);
CREATE INDEX IF NOT EXISTS idx_sessions_plan         ON tracker.workout_sessions(workout_plan_id);
CREATE INDEX IF NOT EXISTS idx_sessions_status       ON tracker.workout_sessions(status);
CREATE INDEX IF NOT EXISTS idx_session_items_session ON tracker.workout_session_items(session_id);
CREATE INDEX IF NOT EXISTS idx_workout_sched_user    ON tracker.user_workout_schedule(user_id);
CREATE INDEX IF NOT EXISTS idx_workout_sched_date    ON tracker.user_workout_schedule(scheduled_date);
CREATE INDEX IF NOT EXISTS idx_meal_sched_user       ON tracker.user_meal_schedule(user_id);
CREATE INDEX IF NOT EXISTS idx_meal_sched_date       ON tracker.user_meal_schedule(scheduled_date);
CREATE INDEX IF NOT EXISTS idx_daily_logs_user       ON tracker.daily_logs(user_id);
CREATE INDEX IF NOT EXISTS idx_daily_logs_date       ON tracker.daily_logs(date);

CREATE TABLE IF NOT EXISTS system.notifications (
    id         UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id    UUID NOT NULL REFERENCES accounts.users(id) ON DELETE CASCADE,
    title      VARCHAR NOT NULL,
    message    TEXT NOT NULL,
    type       VARCHAR NOT NULL CHECK (type IN ('info', 'warning', 'success', 'alert')),

    table_name VARCHAR,
    old_values JSONB,
    new_values JSONB,
    timestamp  TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Indexes
CREATE INDEX IF NOT EXISTS idx_notifications_user ON system.notifications(user_id);
CREATE INDEX IF NOT EXISTS idx_notifications_read ON system.notifications(is_read);
CREATE INDEX IF NOT EXISTS idx_audit_actor        ON system.audit_logs(actor_id);
CREATE INDEX IF NOT EXISTS idx_audit_timestamp    ON system.audit_logs(timestamp);
CREATE INDEX IF NOT EXISTS idx_audit_table        ON system.audit_logs(table_name);

CREATE OR REPLACE FUNCTION profile.handle_new_user()
RETURNS TRIGGER AS $$
BEGIN
  INSERT INTO profile.users (id, full_name, role)
  VALUES (
    NEW.id,
    NEW.raw_user_meta_data->>'full_name',
    COALESCE(NEW.raw_user_meta_data->>'role', 'general')
  );
  RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE FUNCTION profile.handle_new_user();

SELECT
  tc.table_schema,
  tc.table_name,
  kcu.column_name,
  ccu.table_schema AS foreign_schema,
  ccu.table_name   AS foreign_table
FROM information_schema.table_constraints AS tc
JOIN information_schema.key_column_usage AS kcu
  ON tc.constraint_name = kcu.constraint_name
JOIN information_schema.constraint_column_usage AS ccu
  ON ccu.constraint_name = tc.constraint_name
WHERE tc.constraint_type = 'FOREIGN KEY'
  AND ccu.table_name = 'users'
ORDER BY tc.table_schema, tc.table_name;


SELECT
  tc.table_schema,
  tc.table_name,
  kcu.column_name,
  ccu.table_schema AS foreign_schema,
  ccu.table_name   AS foreign_table,
  ccu.column_name  AS foreign_column
FROM information_schema.table_constraints AS tc
JOIN information_schema.key_column_usage AS kcu
  ON tc.constraint_name = kcu.constraint_name
JOIN information_schema.constraint_column_usage AS ccu
  ON ccu.constraint_name = tc.constraint_name
WHERE tc.constraint_type = 'FOREIGN KEY'
  AND tc.table_schema IN ('profile', 'plans', 'tracker', 'system')
ORDER BY tc.table_schema, tc.table_name;


SELECT
  ns.nspname   AS table_schema,
  cls.relname  AS table_name,
  att.attname  AS column_name,
  nsp.nspname  AS foreign_schema,
  rel.relname  AS foreign_table,
  con.conname  AS constraint_name
FROM pg_constraint con

CREATE TABLE IF NOT EXISTS plans.routine_exercises (
    id                UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    routine_id        UUID NOT NULL REFERENCES plans.routines(id) ON DELETE CASCADE,
    exercise_id       UUID NOT NULL REFERENCES library.exercises(id) ON DELETE RESTRICT,
    position          INT NOT NULL DEFAULT 0,          -- ordering within the routine
    sets              INT CHECK (sets > 0),
    reps              INT CHECK (reps > 0),            -- target reps per set

-- A4. Create plans.workout_plan_routines
--     Links routines into a workout plan AND defines the schedule slot
--
--     Schedule supports TWO modes (set schedule_type on the parent plan):
--       'nday'  — day_number is 1,2,3,… (n-day cycle, e.g. Day1/Day2/Day3)
--       'weekly'— day_of_week is 0-6 (0=Sunday … 6=Saturday)
--     Only one column is populated depending on the mode.

CREATE TABLE IF NOT EXISTS plans.workout_plan_routines (
    id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    workout_plan_id UUID NOT NULL REFERENCES plans.workout_plans(id) ON DELETE CASCADE,
    routine_id    UUID NOT NULL REFERENCES plans.routines(id) ON DELETE RESTRICT,

    -- schedule slot (fill ONE depending on schedule_type)
    day_number    INT CHECK (day_number >= 1),      -- for 'nday'  mode: 1,2,3,…
    day_of_week   INT CHECK (day_of_week BETWEEN 0 AND 6), -- for 'weekly' mode: 0=Sun…6=Sat
    is_rest_day   BOOLEAN NOT NULL DEFAULT FALSE,   -- mark a day as rest (no routine needed)

    position      INT NOT NULL DEFAULT 0,           -- display order if multiple routines same day

    CONSTRAINT one_schedule_type CHECK (
        (day_number IS NOT NULL AND day_of_week IS NULL)
        OR (day_number IS NULL AND day_of_week IS NOT NULL)
        OR (is_rest_day = TRUE)   -- rest day needs neither
    )
);

CREATE INDEX IF NOT EXISTS idx_plan_routines_plan    ON plans.workout_plan_routines(workout_plan_id);
CREATE INDEX IF NOT EXISTS idx_plan_routines_routine ON plans.workout_plan_routines(routine_id);


-- A5. Add schedule_type to workout_plans so the app knows which mode to render
ALTER TABLE plans.workout_plans
  ADD COLUMN IF NOT EXISTS schedule_type VARCHAR DEFAULT 'nday'
      CHECK (schedule_type IN ('nday', 'weekly'));
